#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <fstream>
#include <cstring>
#include <string>
#include <sstream>
#include <utility> //to use swap()

using namespace std;

static const char* DB_FILE = "letters.dat";


static char* dup_cstr(const char* s)
{
    size_t len = strlen(s);
    char* r = new char[len + 1];
    memcpy(r, s, len + 1);
    return r;
}

static void set_str(char*& dst, const char* src)
{
    delete[] dst;
    dst = dup_cstr(src);
}

static char* read_line_cstr(const char* prompt)
{
    cout << prompt;
    string line;
    getline(cin, line);
    return dup_cstr(line.c_str());
}


class letter
{
private:
    char* adr1;   
    char* fam1;   
    char* name1;  

    char* adr2;   
    char* fam2;   
    char* name2;  

    double cost;  

    static int count; 

public:
    letter();
    letter(const letter& r);
    letter& operator=(const letter& r);
    ~letter();

    void define();
    void display() const;

    static int get_count() { return count; }

    static int search_by_sender(letter** a, int n, const char* sfam2, const char* sname2);

    static void filter_cost_greater(letter** a, int n, double x);

    bool write_bin(ofstream& out) const;
    bool read_bin(ifstream& in);

    double get_cost() const { return cost; }
    const char* get_fam2() const { return fam2; }
    const char* get_name2() const { return name2; }
};

int letter::count = 0;

letter::letter()
{
    adr1 = dup_cstr("");
    fam1 = dup_cstr("");
    name1 = dup_cstr("");
    adr2 = dup_cstr("");
    fam2 = dup_cstr("");
    name2 = dup_cstr("");
    cost = 0.0;
    count++;
}

letter::letter(const letter& r)
{
    adr1 = dup_cstr(r.adr1);
    fam1 = dup_cstr(r.fam1);
    name1 = dup_cstr(r.name1);
    adr2 = dup_cstr(r.adr2);
    fam2 = dup_cstr(r.fam2);
    name2 = dup_cstr(r.name2);
    cost = r.cost;
    count++;
}

letter& letter::operator=(const letter& r)
{
    if (this == &r) return *this;
    set_str(adr1, r.adr1);
    set_str(fam1, r.fam1);
    set_str(name1, r.name1);
    set_str(adr2, r.adr2);
    set_str(fam2, r.fam2);
    set_str(name2, r.name2);
    cost = r.cost;
    return *this;
}

letter::~letter()
{
    delete[] adr1;
    delete[] fam1;
    delete[] name1;
    delete[] adr2;
    delete[] fam2;
    delete[] name2;
    count--;
}

void letter::define()
{
    char* t;

    t = read_line_cstr("adr1 (receiver address): "); set_str(adr1, t); delete[] t;
    t = read_line_cstr("fam1 (receiver surname): "); set_str(fam1, t); delete[] t;
    t = read_line_cstr("name1 (receiver name): ");   set_str(name1, t); delete[] t;

    t = read_line_cstr("adr2 (sender address): ");   set_str(adr2, t); delete[] t;
    t = read_line_cstr("fam2 (sender surname): ");   set_str(fam2, t); delete[] t;
    t = read_line_cstr("name2 (sender name): ");     set_str(name2, t); delete[] t;

    cout << "cost (>0): ";
    while (true)
    {
        string line;
        getline(cin, line);

        size_t L = line.find_first_not_of(" \t\r\n");
        size_t R = line.find_last_not_of(" \t\r\n");
        if (L == string::npos)
        {
            cout << "ERROR: enter a positive number: ";
            continue;
        }
        line = line.substr(L, R - L + 1);

        if (line.find('-') != string::npos || line.find('+') != string::npos)
        {
            cout << "ERROR: cost must be > 0: ";
            continue;
        }

        stringstream ss(line);
        double v;
        char extra;
        if (!(ss >> v) || (ss >> extra))
        {
            cout << "ERROR: invalid input. Enter a positive number: ";
            continue;
        }

        if (v <= 0.0)
        {
            cout << "ERROR: cost must be > 0: ";
            continue;
        }

        cost = v;
        break;
    }
}

void letter::display() const
{
    cout << fam1 << " " << name1 << " | " << adr1
        << " || "
        << fam2 << " " << name2 << " | " << adr2
        << " || cost=" << cost << endl;
}


int letter::search_by_sender(letter** a, int n, const char* sfam2, const char* sname2)
{
    int found = 0;
    for (int i = 0; i < n; ++i)
    {
        if (strcmp(a[i]->get_fam2(), sfam2) == 0 &&
            strcmp(a[i]->get_name2(), sname2) == 0)
        {
            cout << "[" << i << "] ";
            a[i]->display();
            found++;
        }
    }
    if (!found) cout << "NOT FOUND\n";
    return found;
}


void letter::filter_cost_greater(letter** a, int n, double x)
{
    int found = 0;
    for (int i = 0; i < n; ++i)
    {
        if (a[i]->get_cost() > x)
        {
            cout << "[" << i << "] ";
            a[i]->display();
            found++;
        }
    }
    if (!found) cout << "(none)\n";
}


bool letter::write_bin(ofstream& out) const
{
    auto write_str = [&](const char* s)
        {
            int len = (int)strlen(s);
            out.write((char*)&len, sizeof(len));
            out.write(s, len);
        };

    write_str(adr1);
    write_str(fam1);
    write_str(name1);
    write_str(adr2);
    write_str(fam2);
    write_str(name2);
    out.write((char*)&cost, sizeof(cost));
    return true;
}

bool letter::read_bin(ifstream& in)
{
    auto read_str = [&](char*& s) -> bool
        {
            int len;
            if (!in.read((char*)&len, sizeof(len))) return false;
            if (len < 0 || len > 1000000) return false;

            char* tmp = new char[len + 1];
            if (len > 0 && !in.read(tmp, len)) { delete[] tmp; return false; }
            tmp[len] = '\0';

            set_str(s, tmp);
            delete[] tmp;
            return true;
        };

    if (!read_str(adr1)) return false;
    if (!read_str(fam1)) return false;
    if (!read_str(name1)) return false;
    if (!read_str(adr2)) return false;
    if (!read_str(fam2)) return false;
    if (!read_str(name2)) return false;

    if (!in.read((char*)&cost, sizeof(cost))) return false;
    if (cost <= 0.0) return false;

    return true;
}


class DB
{
    letter** a;
    int n;
    int cap;
public:

    void clear() //adding clear function
    {
        for (int i = 0; i < n; ++i) delete a[i];
        n = 0;
    }

    void push_ptr(letter* p)
    {
        reserve(n + 1);
        a[n] = p;
        n++;
    }
    DB() : a(nullptr), n(0), cap(0) {}

    ~DB()
    {
        for (int i = 0; i < n; ++i) delete a[i];
        delete[] a;
    }

    void reserve(int need)
    {
        if (cap >= need) return;
        int nc = (cap == 0 ? 4 : cap);
        while (nc < need) nc *= 2;

        letter** t = new letter * [nc];
        for (int i = 0; i < n; ++i) t[i] = a[i];

        delete[] a;
        a = t;
        cap = nc;
    }

    void push_back(const letter& x)
    {
        reserve(n + 1);
        a[n] = new letter(x);
        n++;
    }

    int get_n() const { return n; } //getting function (public)
    letter** get_a() { return a; }
    letter* const* get_a() const { return a; }


};

static void sort_by_cost(DB& db)
{
    for (int i = 0; i < db.get_n() - 1; ++i)
        for (int j = 0; j < db.get_n() - 1 - i; ++j)
            if (db.get_a()[j]->get_cost() > db.get_a()[j + 1]->get_cost())
                swap(db.get_a()[j], db.get_a()[j + 1]);
}

static void print_all(const DB& db)
{
    cout << "n=" << db.get_n() << "  (existing objects=" << letter::get_count() << ")\n";
    for (int i = 0; i < db.get_n(); ++i)
    {
        cout << "[" << i << "] ";
        db.get_a()[i]->display();
    }
    if (db.get_n() == 0) cout << "(empty)\n";
}


int main()
{
    DB db;

    ifstream fin(DB_FILE, ios::binary);
    if (fin)
    {
        int n;
        if (!fin.read((char*)&n, sizeof(n)) || n < 0 || n > 100000)
        {
            cout << "Load error. Start empty.\n";
            fin.close();
        }
        else
        {
            db.clear(); // clear database before loading new data
            db.reserve(n); // reserve memory for letters from file

            for (int i = 0; i < n; ++i)
            {
                letter* x = new letter();
                if (x->read_bin(fin))
                {
                    db.push_ptr(x);
                }
                else
                {
                    delete x;
                    cout << "Load error. Start empty.\n";
                    db.clear();
                    break;
                }
            }

            fin.close();
            if (db.get_n() > 0) cout << "Loaded from disk.\n";
        }
    }
    else
    {
        int k;
        cout << "First run. How many letters to input? ";
        if (!(cin >> k)) return 0;
        cin.ignore(1000000, '\n');

        for (int i = 0; i < k; ++i)
        {
            cout << "\nLetter #" << (i + 1) << endl;
            letter x;
            x.define();
            db.push_back(x);
        }
    }

    while (true)
    {
        cout << "\n==================== DATABASE MENU (Registered Letters) ====================\n"
            << "1) Print entire database\n"
            << "2) Add new letters\n"
            << "3) Search by sender (fam2 + name2)  [MK]\n"
            << "4) Filter letters with cost > X      [DF]\n"
            << "5) Sort letters by cost (ascending order)\n"
            << "6) Save database and exit\n"
            << "=========================================================================\n"
            << "Enter your choice (1-6): ";

        int ch;
        if (!(cin >> ch)) break;
        cin.ignore(1000000, '\n');

        if (ch == 1)
        {
            print_all(db);
        }
        else if (ch == 2)
        {
            int k;
            cout << "How many new letters to add? ";
            cin >> k;
            cin.ignore(1000000, '\n');

            for (int i = 0; i < k; ++i)
            {
                cout << "\nNew letter #" << (i + 1) << endl;
                letter x;
                x.define();
                db.push_back(x);
            }
        }
        else if (ch == 3)
        {
            cout << "\n--- SEARCH FUNCTION [MK]: Sender (fam2 + name2) ---\n";
            char* f = read_line_cstr("Enter fam2 (sender surname): ");
            char* n = read_line_cstr("Enter name2 (sender name): ");
            letter::search_by_sender(db.get_a(), db.get_n(), f, n);
            delete[] f;
            delete[] n;
        }
        else if (ch == 4)
        {
            cout << "\n--- FILTER FUNCTION [DF]: Cost greater than X ---\n";
            cout << "Enter X: ";

            double x;
            while (true)
            {
                string line;
                getline(cin, line);

                size_t L = line.find_first_not_of(" \t\r\n");
                size_t R = line.find_last_not_of(" \t\r\n");
                if (L == string::npos) { cout << "ERROR. Enter X: "; continue; }
                line = line.substr(L, R - L + 1);

                stringstream ss(line);
                char extra;
                if (!(ss >> x) || (ss >> extra))
                {
                    cout << "ERROR. Enter X: ";
                    continue;
                }
                break;
            }

            letter::filter_cost_greater(db.get_a(), db.get_n(), x);
        }
        else if (ch == 5)
        {
            cout << "\n--- SORTING: By cost (ascending) ---\n";
            sort_by_cost(db);
            cout << "Sorted.\n";
        }
        else if (ch == 6)
        {
            ofstream fout(DB_FILE, ios::binary);
            if (!fout)
            {
                cout << "Save ERROR.\n";
                break;
            }

            int nn = db.get_n();
            fout.write((char*)&nn, sizeof(nn));
            for (int i = 0; i < nn; ++i)
                db.get_a()[i]->write_bin(fout);

            fout.close();
            cout << "Saved.\n";
            break;
        }
        else
        {
            cout << "Invalid choice. Please enter a number from 1 to 6.\n";
        }
    }

    return 0;
}
